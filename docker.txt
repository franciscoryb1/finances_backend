-- Crear bbdd con init.sql

Powershell: Get-Content .\init.sql | docker exec -i finances_postgres psql -U admin -d finances_db
cmd /c "docker exec -i finances_postgres psql -U admin -d finances_db < init.sql"


-- Verificar que este todo creado:
docker exec -it finances_postgres psql -U admin -d finances_db -c "\dt"


-- Ver contenedores corriendo:
docker ps
-- Ver todos los contenedores (incluso los detenidos):
docker ps -a
- Ver imágenes disponibles
docker images

-- Levantar tu contenedor con docker-compose
docker-compose up -d
-d significa detached mode → corre en segundo plano.

-- Detener todos los contenedores
docker-compose down

-- Detener contenedor especifico
docker stop finances_postgres

-- Reiniciar el contenedor
docker restart finances_postgres

-- Ejecutar consulta SQL
docker exec -it finances_postgres psql -U admin -d finances_db -c "ALTER TABLE banks ADD COLUMN user_id INT REFERENCES users(id) ON DELETE CASCADE;"
docker exec -it finances_postgres → ejecuta dentro del contenedor activo.

psql -U admin -d finances_db → abre el cliente de PostgreSQL con el usuario admin y la base finances_db.

-c "..." → ejecuta el comando SQL entre comillas.
